int waitYellow = 2000;

int waitBlue = 3000;

int longW = 1000;

  

void setup() {

  // put your setup code here, to run once:

  pinMode(redLED,OUTPUT);

}

  

void loop() {

  // put your main code here, to run repeatedly:

//...

digitalWrite(redLED,HIGH);

delay(waitRed);

  

digitalWrite(yellowLED,HIGH);

delay(waitYellow);

  

digitalWrite(blueLED,HIGH);

delay(waitBlue);

}
